        indien 'status' == 'registered': 
          wijziging waarde van 'path' 02_processed -> 02_registered

        status != 'registered'
          wijziging waarde van 'path' 02_processed -> 03_processed
          verplaats scan naar nieuwe locatie in 03_processed
